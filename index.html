}); homework.forEach(h => { h.photos = photoMap[h.id] || []; }); return { homework }; } function addHomework(p) { initSheets(); const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(SHEET_HOMEWORK); const id = String(new Date().getTime()); sheet.appendRow([id, p.subject, p.title, p.due, false, false, false, '', new Date().toISOString()]); return { ok: true, id }; } function updateStep(p) { const sheet = getHwSheet(); sheet.getRange(row, col).setValue(p.value === 'true'); return { ok: true }; } function updateNote(p) { const sheet = getHwSheet(); const row = findRowById(sheet, p.id); if (!row) return { error: '見つかりません' }; sheet.getRange(row, 8).setValue(p.note || ''); return { ok: true }; } function deleteHomework(p) { const sheet = getHwSheet(); const row = findRowById(sheet, p.id); if (!row) return { error: '見つかりません' };
